Графика в MATLAB

Простой график


x = 0:pi/100:2*pi;
y = sin(x);

plot(x,y)

Два простых графика


x = linspace(-2*pi,2*pi);
y1 = sin(x);
y2 = cos(x);

plot(x,y1,x,y2)

Задание цвета


x = 0:pi/10:2*pi;
y1 = sin(x);
y2 = sin(x-0.25);
y3 = sin(x-0.5);

plot(x,y1,'g',x,y2,'red',x,y3,'b')

Задание стиля


x = 0:pi/10:2*pi;
y1 = sin(x);
y2 = sin(x-0.25);
y3 = sin(x-0.5);

plot(x,y1,'g',x,y2,'b--o',x,y3,'c*')

Задание размеров


x = -pi:pi/10:pi;
y = tan(sin(x)) - sin(tan(x));
plot(x,y,'--gs',...
    'LineWidth',2,...
    'MarkerSize',10,...
    'MarkerEdgeColor','b',...
    'MarkerFaceColor',[0.5,0.5,0.5])

Легенда


x = linspace(0,pi);
y1 = cos(x);
plot(x,y1)
hold on

y2 = cos(2*x);
plot(x,y2)
legend('cos(x)','cos(2x)')

Функция meshgrid


x = -2:0.25:2;
y = x;
[X,Y] = meshgrid(x);

F = X.*exp(-X.^2-Y.^2);
surf(X,Y,F)

Функция plot3


x = -3:0.1:3;
y = -3:0.1:3;
[X,Y] = meshgrid(x, y);
Z = Y.^2 + X.^2;

plot3(X,Y,Z)

Чуть более наглядный пример


t = 0:pi/50:10*pi;
st = sin(t);
ct = cos(t);

plot3(st,ct,t)

Функция mesh


[X,Y] = meshgrid(-8:.5:8);
R = sqrt(X.^2 + Y.^2) + eps;
Z = sin(R)./R;
mesh(X,Y,Z)

Функция meshc


[X,Y] = meshgrid(-3:.125:3);
Z = peaks(X,Y);
meshc(Z)

Функция surf


[X,Y] = meshgrid(1:0.5:10,1:20);
Z = sin(X) + cos(Y);
surf(X,Y,Z)
